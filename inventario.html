document.addEventListener('DOMContentLoaded', function() { const moedasElemento = document.getElementById('moedas'); let moedas = parseInt(moedasElemento.textContent); // Começando com 100 PO // Função para vender um item function venderItem(preco) { // Atualizar as moedas moedas += preco; moedasElemento.textContent = moedas; } // Selecionar todos os botões de vender e adicionar o evento de clique const botoesVender = document.querySelectorAll('.vender-btn'); botoesVender.forEach(botao => { botao.addEventListener('click', function() { const preco = parseInt(botao.getAttribute('data-preco')); // Remover o item do inventário botao.parentElement.remove(); // Vender o item venderItem(preco); }); }); });
